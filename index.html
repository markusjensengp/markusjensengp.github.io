Anti-rollbars with adjustable stiffness

The anti-rollbars are designed to counteract body roll during cornering by balancing the forces between the left and right suspension components. The code compares the forces on the left and right sides for both the front and rear suspensions. If the force on one side is less than its counterpart on the opposite side, it adjusts the force by multiplying the counterpart's force with a stiffness factor (ranging from 0 to 1). This helps even out the suspension force and reduces body roll. 


void APlayerCar::HandleBodyRoll()
{
    SuspensionFL->GetConstraint().Get()->GetConstraintForce( SuspForceFL, SuspAngForceFL );
    SuspensionFR->GetConstraint().Get()->GetConstraintForce( SuspForceFR, SuspAngForceFR );
    SuspensionFL->GetConstraint().Get()->GetConstraintForce( SuspForceRL, SuspAngForceRL );
    SuspensionFR->GetConstraint().Get()->GetConstraintForce( SuspForceRR, SuspAngForceRR );

    // If right or left suspension force is less than its counterpart, 
    // set this force to its counterparts multiplied by 0-1 (stiffness) 
    // to even out suspension force and counter body roll

    if( SuspForceFL.Z < SuspForceFR.Z )
        SuspForceFL = SuspForceFR * FrontRollBarStiffness;
    else if( SuspForceFR.Z < SuspForceFL.Z )
        SuspForceFR = SuspForceFL * FrontRollBarStiffness;
    
    if( SuspForceRL.Z < SuspForceRR.Z )
        SuspForceRL = SuspForceRR * RearRollBarStiffness;
    else if( SuspForceRR.Z < SuspForceRL.Z )
        SuspForceRR = SuspForceRL * RearRollBarStiffness;
}

Project Summary

I have made a multistroke gestrue recognizing algorithm, using the $1 algorithm as a base.

My solution

At first I had to create an algorithm for resizing and resampling shapes into a set number of points so the recognizing algorithm takes in two similar shapes.

Resampling process

To resample the array into the given number of points (sampling_resolution), I find the total path length by sequentially adding the euclidean distance between the points, and then devide this by sampling_resolution - 1, to get the interval length. Then I add points by interpolating new points along the path defined by the original points, ensuring that the new set of points is evenly spaced along the path.

Resample function: 


void Gesture::Resample(TArray& points)
{
    TArray old_points = points;
    points.Empty();
    points.Init({}, sampling_resolution);
    
    points[0] = FMyPoint(FVector2D(old_points[0].m_point.X, old_points[0].m_point.Y), old_points[0].m_stroke_id);
    int points_added = 1;

    // Calculate the interval length based on the path length and sampling resolution.
    const float interval = PathLength(old_points) / (sampling_resolution - 1.0f);

    float D = 0.0f;  // Distance accumulator.

    for (int i = 1; i < old_points.Num(); i++)
    {
        // Only process points that are in the same stroke.
        if (old_points[i - 1].m_stroke_id == old_points[i].m_stroke_id)
        {
            float distance = Distance(old_points[i - 1], old_points[i]);
            
            // While the accumulated distance plus the current segment distance is 
            // greater than or equal to the interval, it should add points
            if (D + distance >= interval)
            {
                FMyPoint first_point = old_points[i - 1];

                // Add interpolated points to the new points array.
                while (D + distance >= interval)
                {
                    // Calculate the interpolation factor t.
                    float t = FMath::Min(FMath::Max((interval - D) / distance, 0.0f), 1.0f);
                    if (isnan(t)) t = 0.5f;
                    
                    // Add the interpolated point to the new points array.
                    points[points_added++] = FMyPoint(
                        FVector2D(
                            (1.0f - t) * first_point.m_point.X + t * old_points[i].m_point.X,
                            (1.0f - t) * first_point.m_point.Y + t * old_points[i].m_point.Y
                        ),
                        old_points[i].m_stroke_id
                    );

                    distance = D + distance - interval;
                    D = 0.0f;
                    first_point = points[points_added - 1];
                }

                D = distance;
            }
            else
            {
                D += distance;
            }
        }
    }

    // If the last point has not been added, add the last point of the old points array.
    if (points_added == sampling_resolution - 1)
        points[points_added] = old_points.Last();
}

Scale and translate

The next step was to scale and translate the shape so that your drawn shape and the template it compares with, shares the same size and position.

Scale function: 


void Gesture::Scale( TArray< FMyPoint >& points )
{
    float min_x = FLT_MAX, min_y = FLT_MAX;
    float max_x = FLT_MIN, max_y = FLT_MIN;

    // Find the bounding box of the points
    for( int i = 0; i < points.Num(); i++ )
    {
        if( min_x > points[ i ].m_point.X ) min_x = points[ i ].m_point.X;
        if( min_y > points[ i ].m_point.Y ) min_y = points[ i ].m_point.Y;
        if( max_x < points[ i ].m_point.X ) max_x = points[ i ].m_point.X;
        if( max_y < points[ i ].m_point.Y ) max_y = points[ i ].m_point.Y;
    }

    // Calculate the scale based on the maximum dimension of the bounding box
    const float scale = FMath::Max( max_x - min_x, max_y - min_y );

    TArray< FMyPoint > old_points = points;
    points.Empty();

    // Scale the points to fit within a unit square
    for( int i = 0; i < old_points.Num(); i++ )
    {
        points.Add( FMyPoint(
            FVector2D( ( old_points[ i ].m_point.X - min_x ) / scale, ( old_points[ i ].m_point.Y - min_y ) / scale ),
            old_points[ i ].m_stroke_id ) );
    }
}

Translating the shape to be centered at coordinates (0, 0), I first had to find the centroid of the shape.

Centroid function: 


FMyPoint Gesture::Centroid( TArray< FMyPoint >& points )
{
    float centroid_x = 0.0f, centroid_y = 0.0f;

    for( int i = 0; i < points.Num(); i++ )
    {
        centroid_x += points[ i ].m_point.X;
        centroid_y += points[ i ].m_point.Y;
    }

    return FMyPoint( FVector2D( centroid_x / points.Num(), centroid_y / points.Num() ), 0 );
}

Translate function: 


// Moves shape to coordinates (0, 0)
void Gesture::TranslateTo( TArray< FMyPoint >& points, FMyPoint centroid )
{
    TArray< FMyPoint > old_points = points;
    points.Empty();

    for( int i = 0; i < old_points.Num(); i++ )
    {
        points.Add( FMyPoint(
            FVector2D( old_points[ i ].m_point.X - centroid.m_point.X, old_points[ i ].m_point.Y - centroid.m_point.Y ),
            old_points[ i ].m_stroke_id ) );
    }
}

Recognition

Finds distance between each point in your drawn shape and a template shape: 


float Recognizer::PathDistance( const TArray< FMyPoint >& path_a, const TArray< FMyPoint >& path_b )
{
    // Number of points in the paths ( path_a and path_b have
    // the same number of points since both are resampled )
    const int n = path_a.Num();

    // Keeps track of which points in path_b are matched
    std::vector< bool > matched( n, false );

    float sum = 0.0f;
    for( int i = 0; i < n; i++ )
    {
        int index = -1;
        float min_distance = FLT_MAX;

        // Finds the closest point in path_b that has not been matched yet
        for( int j = 0; j < n; j++ )
        {
            if( !matched[ j ] )
            {
                const float distance = Gesture::Distance( path_a[ i ], path_b[ j ] );
                if( distance < min_distance )
                {
                    min_distance = distance;
                    index = j;
                }
            }
        }

        // Marks the point in path_b as matched and adds the smallest distance found to sum
        matched[ index ] = true;
        sum += min_distance;
    }

    return ( sum / path_a.Num() );
}

Finds distance at current angle: 


float Recognizer::DistanceAtAngle( const TArray< FMyPoint >& candidate_points, const TArray< FMyPoint >& template_points,
                                   const float& radians )
{
    TArray< FMyPoint > new_points = candidate_points;
    // Rotates the shape by the radians given by the golden section search
    Gesture::RotateBy( new_points, radians );
    return PathDistance( new_points, template_points );
}

Using a golden section search, I could more efficiently narrow down the search range and reduce the number of distance calculations needed to find the optimal angle for matching the shapes. 


void Recognizer::GoldenSectionSearch( const float& a, const float& b, const TArray< FMyPoint >& candidate_points,
                                      const TArray< FMyPoint >& template_points, float& x, float& f )
{
    // Calculate next angle using the golden section ratio
    x = phi * a + ( 1.0 - phi ) * b;
    // Calculate distance at this angle
    f = DistanceAtAngle( candidate_points, template_points, x );
}

Using the GoldenSectionSearch function, DistanceAtAngle finds the best angle and returns the smalles distance between the points. 


float Recognizer::DistanceAtBestAngle( const TArray< FMyPoint >& candidate_points,
                                       const TArray< FMyPoint >& template_points, const float& from_angle,
                                       const float& to_angle, const float& precision )
{
    float current_from_angle = from_angle;
    float current_to_angle = to_angle;
    float x1, x2, d1, d1;

    GoldenSectionSearch( current_from_angle, current_to_angle, candidate_points, template_points, x1, d1 );
    GoldenSectionSearch( current_to_angle, current_from_angle, candidate_points, template_points, x2, d1 );

    while( FMath::Abs( current_to_angle - current_from_angle ) > precision )
    {
        // Update the bounds based on the results of the golden section search
        if( d1 < d2 )
        {
            current_to_angle = x2; // Update the upper bound
            x2 = x1;               // Move the second angle to the first angle
            d2 = d1;               // Move the second distance to the first distance

            // New search
            GoldenSectionSearch( current_from_angle, current_to_angle, candidate_points, template_points, x1, d1 );
        }
        else
        {
            current_from_angle = x1; // Update the lower bound
            x1 = x2;                 // Move the first angle to the second angle
            d1 = d2;                 // Move the first distance to the second distance
            
            // New search
            GoldenSectionSearch( current_to_angle, current_from_angle, candidate_points, template_points, x2, d2 );
        }
    }

    // Return the smallest distance found
    return FMath::Min( d1, d2 );
}

Recognize function (called when a shape is submitted): 


FString Recognizer::Recognize( const TArray< FMyPoint >& candidate_points )
{
    // Ensures that the shape is neither too small nor too large
    if( Gesture::PathLength( candidate_points ) < 25.0f )
        return "Too short!";
    if( Gesture::PathLength( candidate_points ) > 2000.0f )
        return "Too long!";

    // Initializes a gesture to resample, scale and translate your drawn points
    const Gesture gesture = Gesture( candidate_points );

    float min_distance = FLT_MAX;
    FString result = "No spell matched!";

    for( int i = 0; i < m_templates.Num(); i++ )
    {
        // Finds the closest distance between each point in both shapes, angle_range set to 180 gives 
        // a 360 degree range to draw your shape
        const float distance = DistanceAtBestAngle( gesture.m_points, m_templates[ i ].m_points, -angle_range,
                                                    angle_range, angle_precision );
        if( distance < min_distance )
        {
            min_distance = distance;
            result = m_templates[ i ].m_name;
        }
    }
    
    // Calculate a score to use for a margin of error
    const float score = 1.0f - min_distance * 2.0f;
    if( score < 0.8f )
        result = "No spell matched!";

    // Returns the matched template's name as a string
    return result;
}

Adds a template to the data table and the recognizer object: 


void ASpellDungeonPlayer::AddTemplateToDataTable( const FString Name )
{
    FTemplatesDataTable data_row;

    data_row.m_name = Name;
    data_row.m_points = m_painting_widget->GetPoints();

    const FString ContextString = "Templates";
    TArray< FTemplatesDataTable* > rows;

    m_template_table->GetAllRows< FTemplatesDataTable >( ContextString, rows );

    const int row_num = rows.Num() + 1;

    // Adds template to an the templates data table
    m_template_table->AddRow( FName( *FString::FromInt( row_num ) ), data_row );

    // Adds template to an array in the recognizer object
    m_recognizer->AddTemplate( Name, m_painting_widget->GetPoints() );

    m_training_widget->RemoveFromParent();
    m_painting_widget->RemoveFromParent();
    m_player_controller->SetShowMouseCursor( false );
    UWidgetBlueprintLibrary::SetInputMode_GameOnly( GetWorld()->GetFirstPlayerController() );
}

Loads all saved templates from a data table: 


void ASpellDungeonPlayer::LoadTemplates()
{
    if( m_template_table != nullptr )
    {
        const FString context_string = "Templates";
        TArray< FTemplatesDataTable* > rows;

        m_template_table->GetAllRows< FTemplatesDataTable >( context_string, rows );

        for( int i = 0; i < rows.Num(); i++ )
            m_recognizer->AddTemplate( rows[ i ]->m_name, rows[ i ]->m_points );
    }
}
